18 // # Linear Ringbuffer
19 //
20 // This is an implementation of a ringbuffer that will always expose its contents
21 // as a flat array using the mmap trick. It is mainly useful for interfacing with
22 // C APIs, where this feature can vastly simplify program logic by eliminating all
23 // special case handling when reading or writing data that wraps around the edge
24 // of the ringbuffer.
25 //
26 //
27 // # Data Layout
28 //
29 // From the outside, the ringbuffer contents always look like a flat array
30 // due to the mmap trick:
31 //
32 //
33 // (head) <-- size --> (tail)
34 // v v
35 // /-----------------------------------|-------------------------------\
36 // | buffer area | mmapped clone of buffer area |
37 // \-----------------------------------|-------------------------------/
38 // <---------- capacity ------------->
39 //
40 //
41 // --->(tail) (head) -----~~~~>
42 // v v
43 // /-----------------------------------|-------------------------------\
44 // | buffer area | mmapped clone of buffer area |
45 // \-----------------------------------|-------------------------------/
46 //
47 //
48 // # Usage
49 //
50 // The buffer provides two (pointer, length)-pairs that can be passed to C APIs,
51 // `(write_head(), free_size())` and `(read_head(), size())`.
52 //
53 // The general idea is to pass the appropriate one to a C function expecting
54 // a pointer and size, and to afterwards call `commit()` to adjust the write
55 // head or `consume()` to adjust the read head.
56 //
57 // Writing into the buffer:
58 //
59 // bev::linear_ringbuffer rb;
60 // FILE* f = fopen("input.dat", "r");
61 // ssize_t n = ::read(fileno(f), rb.write_head(), rb.free_size());
62 // rb.commit(n);
63 //
64 // Reading from the buffer:
65 //
66 // bev::linear_ringbuffer rb;
67 // FILE* f = fopen("output.dat", "w");
68 // ssize_t n = ::write(fileno(f), rb.read_head(), rb.size();
69 // rb.consume(n);
70 //
71 // If there are multiple readers/writers, it is the calling code's
72 // responsibility to ensure that the reads/writes and the calls to
73 // produce/consume appear atomic to the buffer, otherwise data loss
74 // can occur.
75 //
76 // # Errors and Exceptions
77 //
78 // The ringbuffer provides two way of initialization, one using exceptions
79 // and one using error codes. After initialization is completed, all
80 // operations on the buffer are `noexcept` and will never return an error.
81 //
82 // To use error codes, the `linear_ringbuffer(delayed_init {})` constructor.
83 // can be used. In this case, the internal buffers are not allocated until
84 // `linear_ringbuffer::initialize()` is called, and all other member function
85 // must not be called before the buffers have been initialized.
86 //
87 // bev::linear_ringbuffer rb(linear_ringbuffer::delayed_init {});
88 // int error = rb.initialize(MIN_BUFSIZE);
89 // if (error) {
90 // [...]
91 // }
92 //
93 // The possible error codes returned by `initialize()` are:
94 //
95 // ENOMEM - The system ran out of memory, file descriptors, or the maximum
96 // number of mappings would have been exceeded.
97 //
98 // EINVAL - The `minsize` argument was 0, or 2*`minsize` did overflow.
99 //
100 // EAGAIN - Another thread allocated memory in the area that was intended
101 // to use for the second copy of the buffer. Callers are encouraged
102 // to try again.
103 //
104 // If exceptions are preferred, the `linear_ringbuffer(int minsize)`
105 // constructor will attempt to initialize the internal buffers immediately and
106 // throw a `bev::initialization_error` on failure, which is an exception class
107 // derived from `std::runtime_error`. The error code as described above is
108 // stored in the `errno_` member of the exception.
109 //
110 //
111 // # Concurrency
112 //
113 // It is safe to be use the buffer concurrently for a single reader and a
114 // single writer, but mutiple readers or multiple writers must serialize
115 // their accesses with a mutex.
116 //
117 // If the ring buffer is used in a single-threaded application, the
118 // `linear_ringbuffer_st` class can be used to avoid paying for atomic
119 // increases and decreases of the internal size.
120 //
121 //
122 // # Implementation Notes
123 //
124 // Note that only unsigned chars are allowed as the element type. While we could
125 // in principle add an arbitrary element type as an additional argument, there
126 // would be comparatively strict requirements:
127 //
128 // - It needs to be trivially relocatable
129 // - The size needs to exactly divide PAGE_SIZE
130 //
131 // Since the main use case is interfacing with C APIs, it seems more pragmatic
132 // to just let the caller cast their data to `void*` rather than supporting
133 // arbitrary element types.
134 //
135 // The initialization of the buffer is subject to failure, and sadly this cannot
136 // be avoided. [1] There are two sources of errors:
137 //
138 // 1) Resource exhaustion. The maximum amount of available memory, file
139 // descriptors, memory mappings etc. may be exceeded. This is similar to any
140 // other container type.
141 //
142 // 2) To allocate the ringbuffer storage, first a memory region twice the
143 // required size is mapped, then it is shrunk by half and a copy of the first
144 // half of the buffer is mapped into the (now empty) second half.
145 // If some other thread is creating its own mapping in the second half after
146 // the buffer has been shrunk but before the second half has been mapped, this
147 // will fail. To ensure success, allocate the buffers before branching into
148 // multi-threaded code.
149 //
150 // [1] Technically, we could use `MREMAP_FIXED` to enforce creation of the
151 // second buffer, but at the cost of potentially unmapping random mappings made
152 // by other threads, which seems much worse than just failing. I've spent some
153 // time scouring the manpages and implementation of `mmap()` for a technique to
154 // make it work atomically but came up empty. If there is one, please tell me.
155 //

369 template<typename T>
370 int linear_ringbuffer_<T>::initialize(size_t minsize) noexcept
371 {
372 #ifdef PAGESIZE
373  static constexpr unsigned int PAGE_SIZE = PAGESIZE;
374 #else
375  static const unsigned int PAGE_SIZE = ::sysconf(_SC_PAGESIZE);
376 #endif
377 
378  // Use `char*` instead of `void*` because we need to do arithmetic on them.
379  unsigned char* addr =nullptr;
380  unsigned char* addr2=nullptr;
381 
382  // Technically, we could also report sucess here since a zero-length
383  // buffer can't be legally used anyways.
384  if (minsize == 0) {
385  errno = EINVAL;
386  return -1;
387  }
388 
389  // Round up to nearest multiple of page size.
390  int bytes = minsize & ~(PAGE_SIZE-1);
391  if (minsize % PAGE_SIZE) {
392  bytes += PAGE_SIZE;
393  }
394 
395  // Check for overflow.
396  if (bytes*2u < bytes) {
397  errno = EINVAL;
398  return -1;
399  }
400 
401  // Allocate twice the buffer size
402  addr = static_cast<unsigned char*>(::mmap(NULL, 2*bytes,
403  PROT_READ | PROT_WRITE, MAP_SHARED | MAP_ANONYMOUS, -1, 0));
404 
405  if (addr == MAP_FAILED) {
406  goto errout;
407  }
408 
409  // Shrink to actual buffer size.
410  addr = static_cast<unsigned char*>(::mremap(addr, 2*bytes, bytes, 0));
411  if (addr == MAP_FAILED) {
412  goto errout;
413  }
414 
415  // Create the second copy right after the shrinked buffer.
416  addr2 = static_cast<unsigned char*>(::mremap(addr, 0, bytes, MREMAP_MAYMOVE,
417  addr+bytes));
418 
419  if (addr2 == MAP_FAILED) {
420  goto errout;
421  }
422 
423  if (addr2 != addr+bytes) {
424  errno = EAGAIN;
425  goto errout;
426  }
427 
428  // Sanity check.
429  *(char*)addr = 'x';
430  assert(*(char*)addr2 == 'x');
431 
432  *(char*)addr2 = 'y';
433  assert(*(char*)addr == 'y');
434 
435  capacity_ = bytes;
436  buffer_ = addr;
437 
438  return 0;
439 
440 errout:
441  int error = errno;
442  // We actually have to check for non-null here, since even if `addr` is
443  // null, `bytes` might be large enough that this overlaps some actual
444  // mappings.
445  if (addr) {
446  ::munmap(addr, bytes);
447  }
448  if (addr2) {
449  ::munmap(addr2, bytes);
450  }
451  errno = error;
452  return -1;
453 }
