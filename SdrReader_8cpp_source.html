93  return false;

119 auto SdrReader::tune(uint32_t frequency, uint32_t sample_rate,
120  uint32_t bandwidth, double gain, const std::string& antenna) -> bool {
121  _frequency = frequency;
122  _filterBw = bandwidth;
123 
124  _buffer.clear();
125  _high_watermark_reached = false;
126 
127  if (_reading_from_file) {
128  _sampleRate = sample_rate;
129  return true;
130  }
131 
132  if (_sdr == nullptr) {
133  return false;
134  }
135 
136  spdlog::info("Tuning to {} MHz, filter bandwidth {} MHz, sample rate {}, gain {}, antenna path {}",
137  frequency/1000000.0, bandwidth/1000000.0, sample_rate/1000000.0, gain, antenna);
138 
139  auto sdr = (SoapySDR::Device*)_sdr;
140 
141  auto antenna_list = sdr->listAntennas(SOAPY_SDR_RX, 0);
142  if (std::find(antenna_list.begin(), antenna_list.end(), antenna) != antenna_list.end()) {
143  sdr->setAntenna( SOAPY_SDR_RX, 0, antenna);
144  _antenna = sdr->getAntenna( SOAPY_SDR_RX, 0);
145  } else {
146  spdlog::error("Unknown antenna \"{}\". Available: {}.", antenna, boost::algorithm::join(antenna_list, ", ") );
147  return false;
148  }
149 
150  if (sdr->hasGainMode(SOAPY_SDR_RX, 0)) {
151  spdlog::info("Disabling AGC");
152  sdr->setGainMode(SOAPY_SDR_RX, 0, false);
153  }
154  auto gain_range = sdr->getGainRange(SOAPY_SDR_RX, 0);
155  _min_gain = gain_range.minimum();
156  _max_gain = gain_range.maximum();
157  if (gain >= gain_range.minimum() && gain <= gain_range.maximum()) {
158  sdr->setGain( SOAPY_SDR_RX, 0, gain);
159  _gain = sdr->getGain( SOAPY_SDR_RX, 0);
160  } else {
161  spdlog::error("Invalid gain setting {}. Allowed range is: {} - {}.", gain, gain_range.minimum(), gain_range.maximum());
162  return false;
163  }
164 
165  sdr->setFrequency( SOAPY_SDR_RX, 0, frequency);
166  sdr->setBandwidth( SOAPY_SDR_RX, 0, bandwidth);
167  sdr->setSampleRate( SOAPY_SDR_RX, 0, sample_rate);
168 
169  _frequency = sdr->getFrequency( SOAPY_SDR_RX, 0);
170  bandwidth = sdr->getBandwidth( SOAPY_SDR_RX, 0);
171  _sampleRate = sdr->getSampleRate( SOAPY_SDR_RX, 0);
172 
173  spdlog::info("SDR tuned to {} MHz, filter bandwidth {} MHz, sample rate {}, gain {}, antenna path {}",
174  _frequency/1000000.0, bandwidth/1000000.0, _sampleRate/1000000.0, _gain, _antenna);
175 
176 
177  auto sensors = sdr->listSensors();
178  if (std::find(sensors.begin(), sensors.end(), "lms7_temp") != sensors.end()) {
179  _temp_sensor_available = true;
180  _temp_sensor_key = "lms7_temp";
181  }
182 
183  return true;
184 }

186 void SdrReader::start() {
187  if (_sdr != nullptr) {
188  auto sdr = (SoapySDR::Device*)_sdr;
189  _stream = sdr->setupStream( SOAPY_SDR_RX, SOAPY_SDR_CF32);
190  if( _stream == nullptr)
191  {
192  spdlog::error("Failed to set up RX stream");
193  SoapySDR::Device::unmake( sdr );
194  return ;
195  }
196  sdr->activateStream( (SoapySDR::Stream*)_stream, 0, 0, 0);
197  }
198  _running = true;
199 
200  // Start the reader thread and elevate its priority to realtime
201  _readerThread = std::thread{&SdrReader::read, this};
202  struct sched_param thread_param = {};
203  thread_param.sched_priority = 50;
204  _cfg.lookupValue("modem.sdr.reader_thread_priority_rt", thread_param.sched_priority);
205 
206  spdlog::debug("Launching sample reader thread with realtime scheduling priority {}", thread_param.sched_priority);
207 
208  int error = pthread_setschedparam(_readerThread.native_handle(), SCHED_RR, &thread_param);
209  if (error != 0) {
210  spdlog::warn("Cannot set reader thread priority to realtime: {}. Thread will run at default priority with a high probability of dropped samples and loss of synchronisation.", strerror(error));
211  }
212 }

227 void SdrReader::read() {
228  std::array<void*, SRSLTE_MAX_CHANNELS> radio_buffers = { nullptr };
229  while (_running) {
230  int toRead = ceil(_sampleRate / 1000.0);
231  if (_buffer.free_size() < toRead * sizeof(cf_t)) {
232  spdlog::debug("ringbuffer overflow");
233  std::this_thread::sleep_for(std::chrono::microseconds(1000));
234  } else {
235  int read = 0;
236  if (_reading_from_file) {
237  std::chrono::steady_clock::time_point entered = {};
238  entered = std::chrono::steady_clock::now();
239  int64_t required_time_us = (1000000.0/_sampleRate) * toRead;
240 
241  radio_buffers[0] = _buffer.write_head();
242  read = srslte_filesource_read_multi(&file_source, radio_buffers.data(), toRead, 1);
243  if ( read == 0 ) {
244  srslte_filesource_seek(&file_source, 0);
245  }
246 
247  if (read> 0) {
248  _buffer.commit( read * sizeof(cf_t) );
249  }
250 
251  std::chrono::microseconds sleep = (std::chrono::microseconds(required_time_us) -
252  std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::steady_clock::now() - entered));
253  std::this_thread::sleep_for(sleep);
254  } else {
255  auto sdr = (SoapySDR::Device*)_sdr;
256  int flags;
257  long long time_ns;
258  void *buffs[] = {_buffer.write_head()}; // NOLINT
259 
260  read = sdr->readStream( (SoapySDR::Stream*)_stream, buffs, toRead, flags, time_ns);
261 
262  if (read> 0) {
263  if (_writing_to_file && _write_samples) {
264  srslte_filesink_write(&file_sink, _buffer.write_head(), read);
265  }
266  _buffer.commit( read * sizeof(cf_t) );
267  }
268  }
269  }
270  }
271  spdlog::debug("Sample reader thread exited");
272 }

274 auto SdrReader::getSamples(cf_t* data, uint32_t nsamples,
275  srslte_timestamp_t *
276  /*rx_time*/) -> int {
277  std::chrono::steady_clock::time_point entered = {};
278  entered = std::chrono::steady_clock::now();
279 
280  int64_t required_time_us = (1000000.0/_sampleRate) * nsamples;
281  size_t cnt = nsamples * sizeof(cf_t);
282 
283  if (_high_watermark_reached && _buffer.size() < (_sampleRate / 1000.0) * 10 * sizeof(cf_t)) {
284  _high_watermark_reached = false;
285  }
286 
287  if (!_high_watermark_reached) {
288  spdlog::debug("size {}", _buffer.size() );
289  while (_buffer.size() < (_sampleRate / 1000.0) * (_buffer_ms / 2.0) * sizeof(cf_t)) {
290  std::this_thread::sleep_for(std::chrono::microseconds(500));
291  }
292  spdlog::debug("Filled ringbuffer to half capacity");
293  _high_watermark_reached = true;
294  }
295 
296  memcpy(data, _buffer.read_head(), cnt);
297  _buffer.consume(cnt);
298 
299  if (_buffer.size() < (_sampleRate / 1000.0) * (_buffer_ms / 2.0) * sizeof(cf_t)) {
300  required_time_us += 500;
301  } else {
302  required_time_us -= 500;
303  }
304 
305  spdlog::debug("read {} samples, adjusted required {} us, delta {} us, sleep adj {}, sleeping for {} us",
306  nsamples,
307  std::chrono::microseconds(required_time_us).count(),
308  std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::steady_clock::now() - _last_read).count(),
309  _sleep_adjustment,
310  (std::chrono::microseconds(_sleep_adjustment + required_time_us) - std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::steady_clock::now() - _last_read)).count());
311 
312  std::chrono::microseconds sleep = (std::chrono::microseconds(_sleep_adjustment + required_time_us) - std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::steady_clock::now() - _last_read));
313 
314  if (sleep.count() > 0) {
315  std::this_thread::sleep_for(sleep);
316  _sleep_adjustment = 0;
317  } else if (sleep.count() > -100000) {
318  _sleep_adjustment = sleep.count();
319  }
320 
321  _last_read = std::chrono::steady_clock::now();
322  return 0;
323 }
